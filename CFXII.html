public int removeDuplicates(int[] nums) {
if (nums.length <= 2) return nums.length;
int p = 2;
for (int i = 2; i < nums.length; i++) {
// 检查实参：nums[p-2] 和 nums[i]
if (nums[p - 2] != nums[i]) {
nums[p] = nums[i];
p++;
}
}
return p;
}
